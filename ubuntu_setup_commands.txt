
# BSC Sniper Bot 2.0 - Ubuntu Setup Commands (Python 3.12)

# Step 1: Make deployment script executable and run it
chmod +x ubuntu_deploy_py312.sh
./ubuntu_deploy_py312.sh

# Step 2: Copy all bot files to your Ubuntu server
# Use one of these methods:

# Method A: SCP (if copying from another machine)
scp bot_files/* ubuntu@your-server-ip:~/bsc_sniper_bot_2/

# Method B: Manual creation on server
cd ~/bsc_sniper_bot_2
# Then create each file manually using nano

# Step 3: Create .env file with your credentials
nano .env
# Add your private key:
# PRIVATE_KEY=your_64_character_hex_private_key_here
# TELEGRAM_BOT_TOKEN=your_telegram_bot_token
# TELEGRAM_CHANNEL_ID=your_telegram_channel_id

# Step 4: Update config.json if needed
nano config.json

# Step 5: Run the bot
source venv/bin/activate
python bsc_sniper_bot_2.py

# Step 6: Run as service (optional - for 24/7 operation)
sudo nano /etc/systemd/system/bsc-sniper.service
# Add the service configuration from setup_guide.md

sudo systemctl daemon-reload
sudo systemctl enable bsc-sniper
sudo systemctl start bsc-sniper
sudo systemctl status bsc-sniper
